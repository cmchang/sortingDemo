    procedure bubbleSort( A : list of sortable items ){
           repeat     
             swapped = false
             for i = 1 to length(A) - 1 inclusive do:
               /* if this pair is out of order */
               if A[i-1] > A[i] then
                 /* swap them and remember something changed */
                 swap( A[i-1], A[i] )
                 swapped = true
               end if
             end for
           until not swapped
        end procedure
        }
